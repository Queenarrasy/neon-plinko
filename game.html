getWinRate() : 30; const isSettinganMenang = (Math.random() * 100) <= winRate; const ball = document.createElement('div'); ball.className = 'ball'; // Start Position dengan variasi sedikit agar tidak kaku let posX = (container.offsetWidth / 2) + (Math.random() * 4 - 2); let posY = 15; let velocityX = 0; let gravity = 0.8; // Efek berat let friction = 0.95; // Gesekan udara ball.style.left = posX + 'px'; ball.style.top = posY + 'px'; container.appendChild(ball); let currentRow = 0; let currentPathIdx = 0; // Melacak posisi relatif terhadap paku const totalRows = 7; const frameRate = 1000 / 60; // 60 FPS untuk gerakan smooth const move = setInterval(() => { // Simulasi Gravitasi & Kecepatan posY += 4.5; // Kecepatan jatuh konstan tiap frame // Cek jika bola mencapai baris paku baru let nextRowY = (currentRow + 2) * 35; if (posY >= nextRowY && currentRow < totalRows) { currentRow++; // Logika Pantulan: Tentukan arah ke kiri (-1) atau kanan (1) let direction; // Logic keberuntungan vs settingan if (isSettinganMenang) { // Cenderung ke arah pinggir (multiplier pink/5x) direction = (currentPathIdx >= 0) ? 1 : -1; } else { // Cenderung ke tengah (multiplier putih/0.2x) direction = (currentPathIdx > 0) ? -1 : 1; } // Tambahkan randomness agar bola "berpencar" tidak monoton if (Math.random() > 0.6) direction *= -1; // Beri dorongan horizontal (Impulse) velocityX = direction * (isTurbo ? 3.5 : 2.5); currentPathIdx += direction; // Efek visual paku bergetar sedikit (optional impact feel) posY -= 2; // Pantulan ke atas sedikit saat kena paku } // Update posisi horizontal dengan physics sederhana posX += velocityX; velocityX *= friction; // Bola melambat setelah memantul ball.style.top = posY + 'px'; ball.style.left = posX + 'px'; // Jika sudah melewati semua baris if (currentRow >= totalRows && posY > (totalRows + 2) * 35 + 10) { clearInterval(move); // Snap bola agar masuk ke tengah kotak const containerWidth = container.offsetWidth; const slotWidth = containerWidth / 7; const finalIdx = Math.max(0, Math.min(6, Math.round(currentPathIdx + 3))); // Animasi masuk ke kotak ball.style.transition = "all 0.3s ease-out"; ball.style.left = `calc(50% + ${(finalIdx - 3) * (containerWidth / 10)}px)`; setTimeout(() => { const ballRect = ball.getBoundingClientRect(); ball.remove(); const winMult = slots[finalIdx]; const totalWin = currentBet * winMult; updateSaldo(totalWin); if(totalWin > 0) { showFloatingWin(totalWin, ballRect.left, ballRect.top); } if(winMult >= 2) { document.body.style.boxShadow = "inset 0 0 50px var(--yellow)"; setTimeout(() => document.body.style.boxShadow = "none", 500); } }, 150); } }, isTurbo ? 10 : 16); // Durasi per frame (Turbo lebih cepat) }
